# JS 高阶编程技巧

# 引言

# 1. 单列设计模式

  • 单例设计模式是最原始的模块思想，如最早AMD、CMD、CommonJS等等
  • 通俗点讲，单利设计模式类似与之前开发一个页面会划分多个功能模块，项目开发的时候讲模块进行划分，每个人负责各自的模块，例如：
// A负责 开发模块 a模块 和 b模块
let newsModuleA = (function () {
  let a = function () {
    console.log('模块a')
  }
  let b = function () {
    console.log('模块b')
  }
  // 利用闭包的特性，可以基于window.xxx把其暴露到全局上（）
  // 缺点：如果全局暴露的东西过多，会存在命名冲突
  // window.a = a();

  // 建议使用return的方式
  return { a }
})()


// B负责 开发模块 c模块 和 d模块
let newsModule2 = (function(){
    newModule.a(); //调用另一个A人开发一个模块的方法

    // B 开发的模块
    function a() {console.log(c);}
    function b() {console.log(c);}
    function c() {console.log(c);}
    function d() {console.log(c);}

    return {
        /**
         * 在单例设计模式的基础上，增加一个命令模式，init作为当前模块业务的入口，
         * 以后只需要执行newsModule2.init()，我们再init中根据业务需求，把所有
         * 编写的方法依次调用执行即可。
         *
         * */

        // 下面的写法等价于：init:function() {}
        init() {
            a();
            b();
            c();
            d();
        };
    }
})()
newsModule2.init()

# 2. 惰性函数

  • 惰性函数的思想也是基于闭包思想，如项目中通过 DOM 绑定事件的兼容性判断处理，最开始很容易会写出下面的方法：
function emit (element, type, callback) {
  if (element.addElementListener) {
    element.addElementListener(type, callback, false)
  } else if (element.attachEvent) {
    element.attachEvent('on' + type, callback)
  } else {
    element['on' + type] = callback
  }
}
  • 上面的写法，会出现一个问题，就在绑定多个的时候，就会发现每一次调用都要判断一下，其实用户在第一次访问的时候，已经判定当前使用环境，所以可以借用惰性函数的思想来优化此方法，如下：
function observerEvent (element, type, func) {
  /**
   * 第一次执行observerEvent，根据兼容判断，重构了
   * 这个函数，重构后的小函数是不需要做兼容验证处理的
   */
  if (element.addEventListener) {
    observerEvent = function (element, type, func) {
      element.addEventListener(type, func)
    }
  } else if (element.attachEvent) {
    observerEvent = function (element, type, func) {
      element.attachEvent('on' + type, func)
    }
  } else {
    observerEvent = function (element, type, func) {
      element['on' + type] = func
    }
  }
  // 第一次也需要执行重构后的方法，实现事件绑定
  observerEvent(element, type, func)
}

# 3. 柯里化函数

  • 预先存储或者预先处理的概念，通俗点就是一个大函数返回小函数
let obj = { x: 100 }
function fn (y) {
  console.log(y)
  this.x += y
  console.log(this)
}
let button = document.getElementById('btn')
button.onclick = function () {
  fn.call(obj, 200)
}
// 点击的时候改变this执行，并把函数fn执行

# 柯里化函数执行的思想

  • 利用闭包的机制，把一些内容事先存储和处理，等到后期需要的时候拿来使用即可
  • bind的实现原理就是利用闭包（柯理化函数编程思想），预先把需要执行的函数的this指向改变和后续需要给函数传递的参数信息等保存到不销毁的上下文中，后续使用的时候直接使用
/**
 * bind: 预先处理内容
 *   @params
 *      func: 要执行的函数
 *      context：需要改变的this指向
 *      args: 给函数传递的参数
 *   @return
 *      返回一个代理函数
 */
function bind (context, ...args) {
  let _this = this
  return function proxy (...inners) {
    //func.call(context, ...args);
    this.apply(context, args.concat(inners)) //这里也会把ev(事件对象)绑定了
  } // proxy 名字没有什么作用，匿名函数具名化
}
  • 预先存储（预先处理）思想，应用如下：
function fn (x, y) {
  /**
   * 第一次执行函数，形成一个临时不被释放的上下文（闭包），
   * 在闭包中我们保存下来传递的参数信息，当后期执行小函数
   * 的时候，可以基于作用域链机制，找到闭包中存储的信息，
   * 并且拿来使用，所以形成的闭包类似于预先把一些信息进行存储
   */
  return function (z) {
    // 最后小函数执行的时候，需要把之前传递的值和最新传递的值进行累加
    return x + y + z
  }
}
let res = fn(1, 2)(3)
console.log(res) //=>6  1+2+3
  • 在 React 的redux、react-redux和Vue中的很多核心源码采用了柯里化函数执行的思想

# 4. compose 函数扁平化

  • 在函数式编程当中有一个很重要的概念就是函数组合， 实际上就是把处理数据的函数像管道一样连接起来， 然后让数据穿过管道得到最终的结果。 例如：
const add1 = x => x + 1
const mul3 = x => x * 3
const div2 = x => x / 2
div2(mul3(add1(add1(0)))) //=>3
  • 而这样的写法可读性明显太差了，我们可以构建一个 compose 函数，它接受任意多个函数作为参数（这些函数都只接受一个参数），然后 compose 返回的也是一个函数，达到以下的效果：
const operate = compose(div2, mul3, add1, add1)
operate(0) //=>相当于 div2(mul3(add1(add1(0))))
operate(2) //=>相当于 div2(mul3(add1(add1(2))))
  • 简而言之：compose 可以把类似于 f(g(h(x)))这种写法简化成 compose(f, g, h)(x)，实现 compose 函数方法如下：
const add1 = x => x + 1
const mul3 = x => x * 3
const div2 = x => x / 2

function compose (...fns) {
  // fns
  return function anonymous (val) {
    // val第一个函数执行时候需要的实参  0
    if (fns.length === 0) return val
    if (fns.length === 1) return funcs[0](val)
    // return fns.reverse().reduce((N, item) => {
    // 	return typeof N === "function" ? item(N(val)) : item(N);
    // });
    return fns.reverse().reduce((N, item) => {
      return item(N)
    }, val)
  }
}
let result = compose(div2, mul3, add1)(5)
console.log(result)

let arr = [10, 20, 30, 40]; 
result = arr.reduce((N, item) => {
	// 第一次：10 20
	// 第二次：30 30
	// 第三次: 60 40
	// ...
    // reduce只传递一个回调函数，那么N第一次默认是第一项，
    // 后续的N是上一次函数执行的处理结果
	// console.log(N, item);
	return N + item;
});
console.log(result); 
result = arr.reduce((N, item) => {
	// console.log(N, item);
	return N + item;
}, 0); //=>REDUCE的第二个参数就是给N赋值的初始值 ITEM从数组第一项开始遍历
console.log(result); 

// redux源码里面的compose函数思想
function compose (...fns) {
  if (fns.length === 0) {
    return arg => arg
  }

  if (fns.length === 1) {
    return fns[0]
  }

  return fns.reduce((a, b) => (...args) => a(b(...args)))
}
